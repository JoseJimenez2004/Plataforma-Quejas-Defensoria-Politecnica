\section*{Introducción}
Este documento especifica los requerimientos no funcionales del Sistema de Gestión DDP, estableciendo atributos de calidad, restricciones y criterios de operación para garantizar el cumplimiento de los procesos institucionales.

\section{Requerimientos No Funcionales por Proceso}

\subsection{Proceso DDP-PO-01: Control y Gestión Administrativa de Documentos}

\begin{tcolorbox}[title=RNF-01-001: Disponibilidad del Proceso, colback=yellow!5!white]
    \begin{itemize}[leftmargin=*]
        \item \textbf{ID:} RNF-01-001
        \item \textbf{Proceso:} DDP-PO-01
        \item \textbf{Categoría:} Disponibilidad
        \item \textbf{Prioridad:} Alta
        \item \textbf{Descripción:} El proceso debe garantizar una atención expedita y oportuna a los miembros de la comunidad politécnica.
        \item \textbf{Métricas:}
        \begin{itemize}
            \item Tiempo máximo de registro: 5 minutos por documento
            \item Disponibilidad del sistema: 99\% en horario laboral
            \item Tiempo de búsqueda: \mleq 30 segundos
            \item Capacidad concurrente: 20 usuarios simultáneos
        \end{itemize}
        \item \textbf{Horario Laboral:} Lunes a Viernes, 8:00 - 18:00 horas
        \item \textbf{Tolerancia:} 1\% de tiempo de inactividad mensual
        \item \textbf{Método de Verificación:} Monitoreo de logs y métricas de uso
        \item \textbf{Requisitos Institucionales:}
        \begin{enumerate}
            \item Atención inmediata a documentos urgentes
            \item Priorización según tipo de documento
            \item Notificaciones en tiempo real
        \end{enumerate}
    \end{itemize}
\end{tcolorbox}

\subsection{Proceso DDP-PO-02: Atención a Solicitudes de Orientación y/o Quejas}

\begin{tcolorbox}[title=RNF-02-001: Confidencialidad de Quejas, colback=orange!5!white]
    \begin{itemize}[leftmargin=*]
        \item \textbf{ID:} RNF-02-001
        \item \textbf{Proceso:} DDP-PO-02
        \item \textbf{Categoría:} Seguridad/Confidencialidad
        \item \textbf{Prioridad:} Crítica
        \item \textbf{Descripción:} La información de las quejas debe tratarse de manera confidencial y reservada bajo la normativa de transparencia.
        \item \textbf{Normativa Aplicable:}
        \begin{itemize}
            \item Ley General de Protección de Datos Personales
            \item Ley Federal de Transparencia
            \item Reglamento Interno del IPN
        \end{itemize}
        \item \textbf{Medidas de Seguridad:}
        \begin{enumerate}
            \item Encriptación de datos sensibles en reposo y tránsito
            \item Control de acceso basado en roles (RBAC)
            \item Registro de auditoría completo (who, what, when)
            \item Anonimización para reportes estadísticos
            \item Borrado seguro según políticas de retención
        \end{enumerate}
        \item \textbf{Niveles de Acceso:}
        \begin{itemize}
            \item Nivel 1: Personal DDP (acceso limitado)
            \item Nivel 2: Subdefensores (acceso por área)
            \item Nivel 3: Titular DDP (acceso completo)
            \item Nivel 4: Administradores (sin acceso a contenido)
        \end{itemize}
        \item \textbf{Método de Verificación:} Auditorías de seguridad trimestrales
    \end{itemize}
\end{tcolorbox}

\subsection{Proceso DDP-PO-03: Acciones de Formación en Derechos Humanos y Politécnicos}

\begin{tcolorbox}[title=RNF-03-001: Capacidad de Recursos, colback=blue!5!white]
    \begin{itemize}[leftmargin=*]
        \item \textbf{ID:} RNF-03-001
        \item \textbf{Proceso:} DDP-PO-03
        \item \textbf{Categoría:} Capacidad
        \item \textbf{Prioridad:} Media
        \item \textbf{Descripción:} Los recursos tecnológicos y didácticos deben ser suficientes para cubrir las exigencias de cada acción formativa específica.
        \item \textbf{Requisitos Técnicos:}
        \begin{itemize}
            \item Soporte para videoconferencias con 100 participantes
            \item Almacenamiento para materiales didácticos (1TB)
            \item Plataforma de aprendizaje compatible con SCORM
            \item Ancho de banda mínimo: 100 Mbps dedicado
        \end{itemize}
        \item \textbf{Recursos Didácticos:}
        \begin{itemize}
            \item Streaming de video en HD (720p mínimo)
            \item Compartición de pantalla
            \item Salas virtuales para trabajos en grupo
            \item Biblioteca digital de materiales
        \end{itemize}
        \item \textbf{Escalabilidad:}
        \begin{itemize}
            \item Soporte para 500 usuarios concurrentes
            \item Capacidad para 50 acciones formativas simultáneas
            \item Almacenamiento escalable según demanda
        \end{itemize}
        \item \textbf{Método de Verificación:} Pruebas de carga y simulación
    \end{itemize}
\end{tcolorbox}

\subsection{Proceso DDP-PO-04: Acciones de Promoción en Derechos Humanos y Politécnicos}

\begin{tcolorbox}[title=RNF-04-001: Eficiencia en Promoción, colback=green!5!white]
    \begin{itemize}[leftmargin=*]
        \item \textbf{ID:} RNF-04-001
        \item \textbf{Proceso:} DDP-PO-04
        \item \textbf{Categoría:} Eficiencia
        \item \textbf{Prioridad:} Media
        \item \textbf{Descripción:} Las acciones de promoción deben sistematizarse para garantizar que se proporcionen de manera eficaz a la comunidad que lo requiera.
        \item \textbf{Métricas de Eficiencia:}
        \begin{itemize}
            \item Tiempo de respuesta a solicitudes: \mleq 48 horas
            \item Tasa de utilización de recursos: \mgeq 85\%
            \item Reducción de trámites manuales: \mgeq 70\%
            \item Satisfacción de usuarios: \mgeq 90\%
        \end{itemize}
        \item \textbf{Procesos a Optimizar:}
        \begin{enumerate}
            \item Solicitud de actividades
            \item Programación de calendario
            \item Confirmación de participantes
            \item Generación de evidencia
            \item Reporte de actividades
        \end{enumerate}
        \item \textbf{Automatización Requerida:}
        \begin{itemize}
            \item Programación automática según disponibilidad
            \item Notificaciones automáticas a participantes
            \item Generación de documentación oficial
            \item Cálculo de indicadores de gestión
        \end{itemize}
        \item \textbf{Método de Verificación:} Análisis de tiempos y procesos
    \end{itemize}
\end{tcolorbox}

\subsection{Proceso DDP-PO-05: Gestión para la Formalización de Convenios de Colaboración}

\begin{tcolorbox}[title=RNF-05-001: Calidad de Medios, colback=purple!5!white]
    \begin{itemize}[leftmargin=*]
        \item \textbf{ID:} RNF-05-001
        \item \textbf{Proceso:} DDP-PO-05
        \item \textbf{Categoría:} Calidad
        \item \textbf{Prioridad:} Alta
        \item \textbf{Descripción:} Los espacios y medios para la gestión (como videoconferencias) deben cumplir con los estándares institucionales.
        \item \textbf{Estándares Institucionales:}
        \begin{itemize}
            \item Normativa de Comunicación Oficial IPN
            \item Estándares de Imagen Institucional
            \item Protocolos de Ceremonial y Protocolo
            \item Formatos Oficiales de Documentación
        \end{itemize}
        \item \textbf{Requisitos de Calidad:}
        \begin{enumerate}
            \item Videoconferencias en calidad Full HD (1080p)
            \item Audio claro sin interferencias
            \item Sala virtual con branding institucional
            \item Documentos en formatos oficiales (.docx, .pdf)
            \item Respaldo de sesiones con calidad de archivo
        \end{enumerate}
        \item \textbf{Equipamiento Mínimo:}
        \begin{itemize}
            \item Cámaras HD con seguimiento automático
            \item Micrófonos de condensador direccionales
            \item Sistema de iluminación profesional
            \item Pantallas de 55" mínimo
        \end{itemize}
        \item \textbf{Método de Verificación:} Checklist de calidad pre-sesión
    \end{itemize}
\end{tcolorbox}
